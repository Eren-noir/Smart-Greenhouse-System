START Smart_Greenhouse_System

 // Initialize tracking variables
    SET consecutive_alerts = 0
    SET critical_flag = FALSE
    
 // Main simulation loop
FOR interval = 1 TO 10
        PRINT "=== Interval " + interval + " ==="
IF interval = 1 THEN
    SET temperature = 25
    SET humidity = 50
    SET light_intensity = 300
    SET soil_moisture = 70
    SET co2_level = 1000

ELSE IF interval = 2 THEN
    // Trigger watering 
    SET temperature = 32
    SET humidity = 45
    SET light_intensity = 500
    SET soil_moisture = 30
    SET co2_level = 600

ELSE IF interval = 3 THEN
    // Trigger light watering 
    SET temperature = 36
    SET humidity = 55
    SET light_intensity = 400
    SET soil_moisture = 40
    SET co2_level = 700

ELSE IF interval = 4 THEN
    //  needs full shading
    SET temperature = 28
    SET humidity = 60
    SET light_intensity = 1200
    SET soil_moisture = 50
    SET co2_level = 500

ELSE IF interval = 5 THEN
    // partial shading
    SET temperature = 26
    SET humidity = 55
    SET light_intensity = 900
    SET soil_moisture = 45
    SET co2_level = 450

ELSE IF interval = 6 THEN
    // First alert 
    SET temperature = 37
    SET humidity = 20
    SET light_intensity = 1150
    SET soil_moisture = 40
    SET co2_level = 800

ELSE IF interval = 7 THEN
    // Second consecutive alert 
    SET temperature = 38
    SET humidity = 22
    SET light_intensity = 1100
    SET soil_moisture = 25
    SET co2_level = 900

ELSE IF interval = 8 THEN
    // Third consecutive alert 
    SET temperature = 39
    SET humidity = 18
    SET light_intensity = 600
    SET soil_moisture = 28
    SET co2_level = 1300

ELSE IF interval = 9 THEN
    // Break alert streak
    SET temperature = 29
    SET humidity = 45
    SET light_intensity = 400
    SET soil_moisture = 55
    SET co2_level = 500

ELSE IF interval = 10 THEN
    // Skip watering 
    SET temperature = 27
    SET humidity = 65
    SET light_intensity = 350
    SET soil_moisture = 75
    SET co2_level = 400

END IF

// Display current sensor readings
PRINT "Temperature: " + temperature + "°C"
PRINT "Humidity: " + humidity + "%"
PRINT "Light Intensity: " + light_intensity + " lux"
PRINT "Soil Moisture: " + soil_moisture + "%"
PRINT "CO2 Level: " + co2_level + " ppm"



// DECISION 1: WATERING CONTROL
SET watering_action = "No watering needed"

IF soil_moisture < 35 AND (humidity < 40 OR temperature > 30)   THEN
    SET watering_action = "Initiate watering"
    
ELSE IF (soil_moisture >= 35 AND soil_moisture <= 50 AND temperature > 35) 
 THEN
    SET watering_action = "Initiate light watering"
    
ELSE IF (soil_moisture > 70) 
 THEN 
    SET watering_action = "Skip watering - soil too wet"
    
END IF

PRINT "Watering Decision: " + watering_action


// DECISION 2: SHADING CONTROL
SET shading_action = "No shading"

SWITCH light_intensity
    CASE < 300:
        SET shading_action = "Open shades"
        
    CASE 300 TO 800:
        SET shading_action = "No shading"
        
    CASE 800 TO 1000:
        SET shading_action = "Close shades partially"
        
    CASE > 1000:
        SET shading_action = "Close the shades fully"
        
END SWITCH

PRINT "Shading Decision: " + shading_action


// DECISION 3: ALERT SYSTEM
SET risk_count = 0
SET alert_triggered = FALSE

IF temperature > 36 THEN
    risk_count = risk_count + 1
END IF

IF humidity < 25 THEN
    risk_count = risk_count + 1
END IF

IF co2 > 1200 THEN
    risk_count = risk_count + 1
END IF

IF soil_moisture < 30 THEN
    risk_count = risk_count + 1
END IF

IF light_intensity > 1100 THEN
    risk_count = risk_count + 1
END IF

// Decide if alert should be raised
IF risk_count >= 3 THEN
    SET alert_triggered = TRUE
    PRINT "ALERT: " + risk_count + " risk factors detected!"
ELSE
    PRINT "Status: Normal (" + risk_count + " risk factors)"
END IF


// DECISION 4: CRITICAL FLAG TRACKING
IF alert_triggered = TRUE THEN
    consecutive_alerts = consecutive_alerts + 1
    
    IF consecutive_alerts > 2 THEN
        SET critical_flag = TRUE
        PRINT "* CRITICAL RISK FLAG RAISED *"
        PRINT "Alerts for " + consecutive_alerts + " consecutive intervals!"
    END IF
    
ELSE
    consecutive_alerts = 0
END IF

// Display current status
PRINT "Consecutive alerts: " + consecutive_alerts
PRINT "Critical flag status: " + critical_flag
PRINT "===================="
PRINT " "

END FOR

// Mock AI
PRINT " "
PRINT "==AI DASHBOARD=="
PRINT "SYSTEM ACTIVITY:"
PRINT "-Total intervals simulated:10"
PRINT "-Watering actions triggered:2"
PRINT "-Critical flag raised:YES"
PRINT "PREDICTIVE ANALYSIS:"
SET recent_soil_avg = (28 + 55 + 75) / 3
PRINT "Soil moisture trend: " + recent_soil_avg + "%"
PRINT "AI RECOMMENDATION:"
IF recent_soil_avg < 40 THEN
    PRINT "Schedule watering in next 2 hours"
ELSIF recent_soil_avg > 60 THEN  
    PRINT "Reduce watering frequency"
ELSE
    PRINT "Continue current watering schedule"
END IF

 
STOP
